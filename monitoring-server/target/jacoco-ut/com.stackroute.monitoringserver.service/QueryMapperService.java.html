package com.stackroute.monitoringserver.service;

import com.stackroute.monitoringserver.domain.ContainerMetrics;
import com.stackroute.monitoringserver.domain.ContainerMetricsMemory;
import com.stackroute.monitoringserver.domain.HealthMetrics;
import com.stackroute.monitoringserver.domain.HealthMetricsDetails;
import org.influxdb.dto.QueryResult;

import java.util.LinkedList;
import java.util.List;

public class QueryMapperService {

    public List<HealthMetrics> healthMapper(QueryResult queryResult){
        List<HealthMetrics> list=new LinkedList<HealthMetrics>();
        if(!queryResult.getResults().isEmpty() && !queryResult.getResults().get(0).getSeries().isEmpty()){

            int i=queryResult.getResults().get(0).getSeries().get(0).getValues().size();
            HealthMetrics healthMetrics=new HealthMetrics();
            HealthMetricsDetails healthMetricsDetails=new HealthMetricsDetails();
            for (int j = 0; j < i; j++) {
                healthMetricsDetails.setDiskFree(
                        queryResult.getResults().get(0).getSeries().get(0).getValues().get(j).get(1).toString()
                );
                healthMetricsDetails.setDiskTotal(
                        queryResult.getResults().get(0).getSeries().get(0).getValues().get(j).get(2).toString()
                );
                healthMetrics.setDetails(healthMetricsDetails);
                healthMetrics.setStatus(
                        queryResult.getResults().get(0).getSeries().get(0).getValues().get(j).get(3).toString()
                );
                list.add(healthMetrics);
            }
        }
        return list;
    }

    public List<ContainerMetrics> containerMapper(QueryResult queryResult){
        List<ContainerMetrics> list= new LinkedList<>();
        if(!queryResult.getResults().isEmpty() && !queryResult.getResults().get(0).getSeries().isEmpty()){

            int i=queryResult.getResults().get(0).getSeries().get(0).getValues().size();
            ContainerMetrics containerMetrics=new ContainerMetrics();
            ContainerMetricsMemory containerMetricsMemory=new ContainerMetricsMemory();
            for (int j = 0; j < i; j++) {
                containerMetrics.setContainer(
                        queryResult.getResults().get(0).getSeries().get(0).getValues().get(j).get(1).toString()
                );
                containerMetricsMemory.setRaw(
                        queryResult.getResults().get(0).getSeries().get(0).getValues().get(j).get(2).toString()
                );
                containerMetricsMemory.setPercent(
                        queryResult.getResults().get(0).getSeries().get(0).getValues().get(j).get(3).toString()
                );
                containerMetrics.setMemory(containerMetricsMemory);
                containerMetrics.setCpu(
                        queryResult.getResults().get(0).getSeries().get(0).getValues().get(j).get(4).toString()
                );
                list.add(containerMetrics);
            }
        }
        return list;
    }
}
